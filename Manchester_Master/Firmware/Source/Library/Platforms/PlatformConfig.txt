Файл PlatformConfig.h, в котором перечислялись все возможные платформы,
перенесен в проект, чтобы в нем перечислить только те платформы,
которые поддерживаются проектом.
Например, для ACPB, файл PlatformConfig.h может выглядеть так:

// Конфигуратор платформы,
// Выбор правильного файла-конфигуратора, в зависимости от настройки препроцессора IAR EWARM.
#if		defined( USE_PLATFORM_LOOCH_619_00_04_00 )
	#include "LOOCH.619.00.04.00\PlatformConfig.h"		// Платформа ЛУЧ.619.00.04.00 (Контроллер трехзондового каверномера АКП(б))
#elif	defined( USE_PLATFORM_LOOCH_623_00_02_00 )
	#include "LOOCH.623.00.02.00\PlatformConfig.h"		// Платформа ЛУЧ.623.00.02.00 (Контроллер однозондового каверномера в составе модулЯ ГГП (гамма-гамма-плотностного))
#elif	defined( USE_PLATFORM_LOOCH_624_00_04_00 )
	#include "LOOCH.624.00.04.00\PlatformConfig.h"		// Платформа ЛУЧ.624.00.04.00 (Контроллер трехзондового каверномера в составе модулЯ ГГП (гамма-гамма-плотностного))
#elif	defined( USE_PLATFORM_LOOCH_513_00_02_00 )
	#include "LOOCH.513.00.02.00\PlatformConfig.h"		// Платформа ЛУЧ.513.00.02.00 (Контроллер трехзондового каверномера длЯ автономного комплекса)
#else
	#error "Select Platform!"
#endif
